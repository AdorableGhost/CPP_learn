#include <iostream>

class Node {
public:
	int data;
	Node* left = nullptr;

class tree {

	Node* Head = nullptr;

	Node* seek_left_last()
	{
		if (tmp->left != nullptr && tmp != nullptr)
		{
			seek_left_last();
			tmp = tmp->left;
		}
		else
		{
			return tmp;
		}
	}

	Node* seek_right_last()
	{
		if (tmp != nullptr && tmp->right != nullptr)
		{
			seek_right_last();
			tmp = tmp->right;
		}
		else
		{
			return tmp;
		}
	}
	  //先序遍历，D-L-R

		if (t == nullptr)

		DLRShow(t->left);
		DLRShow(t->right);

		if (t == nullptr)
			return;

		DLRShow(t->left);
		cout << t->data << "\t" << endl;  		
		DLRShow(t->right);
